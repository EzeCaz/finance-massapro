    15→/* ──────────── Animated Counter ──────────── */
    16→function AnimatedCounter({ end, suffix = '', prefix = '', duration = 2000 }: { end: number; suffix?: string; prefix?: string; duration?: number }) {
    17→  const [count, setCount] = useState(0)
    18→  const ref = useRef<HTMLSpanElement>(null)
    19→  const inView = useInView(ref, { once: true, margin: '-50px' })
    20→
    21→  useEffect(() => {
    22→    if (!inView) return
    23→    let start = 0
    24→    const step = end / (duration / 16)
    25→    const timer = setInterval(() => {
    26→      start += step
    27→      if (start >= end) {
    28→        setCount(end)
    29→        clearInterval(timer)
    30→      } else {
    31→        setCount(Math.floor(start))
    32→      }
    33→    }, 16)
    34→    return () => clearInterval(timer)
    35→  }, [inView, end, duration])
    36→
    37→  return <span ref={ref}>{prefix}{count}{suffix}</span>
    38→}
    39→
    40→/* ──────────── Section Wrapper with InView ──────────── */
    41→function Section({ children, className = '', id }: { children: React.ReactNode; className?: string; id?: string }) {
    42→  const ref = useRef<HTMLElement>(null)
    43→  const inView = useInView(ref, { once: true, margin: '-80px' })
    44→
    45→  return (
    46→    <motion.section
    47→      id={id}
    48→      ref={ref}
    49→      initial={{ opacity: 0, y: 40 }}
    50→      animate={inView ? { opacity: 1, y: 0 } : {}}
    51→      transition={{ duration: 0.7, ease: 'easeOut' }}
    52→      className={className}
    53→    >
    54→      {children}
    55→    </motion.section>
    56→  )
    57→}
